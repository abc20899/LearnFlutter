https://flutter.dev/community/china

export PUB_HOSTED_URL=https://pub.flutter-io.cn
export FLUTTER_STORAGE_BASE_URL=https://storage.flutter-io.cn

export PATH="$PWD/flutter/bin:$PATH"
cd ./flutter
flutter doctor

flutter help
flutter devices
flutter run
flutter format <filename>
flutter build apk
flutter install
